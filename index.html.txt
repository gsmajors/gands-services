        // Helper function for exponential backoff during API calls
        async function fetchWithRetry(url, options, maxRetries = 5) {
            for (let i = 0; i < maxRetries; i++) {
                try {
                    const response = await fetch(url, options);
                    if (response.status !== 429) { // Not a Rate Limit error
                        return response;
                    }
                    const delay = Math.pow(2, i) * 1000 + Math.random() * 1000;
                    await new Promise(resolve => setTimeout(resolve, delay));
                } catch (error) {
                    // Log connection errors but retry
                    const delay = Math.pow(2, i) * 1000 + Math.random() * 1000;
                    await new Promise(resolve => setTimeout(resolve, delay));
                }
            }
            throw new Error("API call failed after multiple retries.");
        }

            const payload = {
                contents: [{ parts: [{ text: userTask }] }],
                systemInstruction: { parts: [{ text: systemPrompt }] },
                generationConfig: {
                    responseMimeType: "application/json",
                    responseSchema: {
                        type: "OBJECT",
                        properties: {
                            "steps": {
                                "type": "ARRAY",
                                "description": "5-7 highly actionable, sequential steps.",
                                "items": { "type": "STRING" }
                            },
                            "emailDraft": {
                                "type": "STRING",
                                "description": "A professional, concise email draft to the client."
                            }
                        },
                        required: ["steps", "emailDraft"]
                    }
                }
            };

            try {
                const response = await fetchWithRetry(API_URL, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(payload)
                });

                const result = await response.json();
                
                if (!response.ok) {
                    throw new Error(result.error?.message || `HTTP error! status: ${response.status}`);
                }

                const jsonText = result.candidates?.[0]?.content?.parts?.[0]?.text;

                if (!jsonText) {
                    throw new Error("Received an empty response from the AI model.");
                }

                const structuredResponse = JSON.parse(jsonText);

                // Display Steps
                if (structuredResponse.steps && Array.isArray(structuredResponse.steps)) {
                    structuredResponse.steps.forEach((step, index) => {
                        const li = document.createElement('div');
                        li.className = 'flex items-start';
                        li.innerHTML = `<span class="font-bold text-brand-accent mr-2">${index + 1}.</span> <span>${step}</span>`;
                        stepsOutput.appendChild(li);
                    });
                } else {
                     stepsOutput.innerHTML = '<p class="text-red-500">Could not parse steps.</p>';
                }

                // Display Email Draft
                if (structuredResponse.emailDraft) {
                    emailOutput.textContent = structuredResponse.emailDraft;
                } else {
                    emailOutput.textContent = 'Could not generate email draft.';
                }
                
                resultsContainer.classList.remove('hidden');

            } catch (e) {
                console.error("Gemini API Error:", e);
                errorDisplay.textContent = `Error strategizing task: ${e.message}. Please try a different task.`;
                errorDisplay.classList.remove('hidden');
            } finally {
                loadingIndicator.classList.add('hidden');
                strategizeButton.disabled = false;
            }
        }

        // Initialize function
        window.onload = () => {
             // Optional: Pre-fill the task input for demonstration
             taskInput.value = "Coordinate the planning and booking process for my company's next off-site retreat.";
        };
    </script>
